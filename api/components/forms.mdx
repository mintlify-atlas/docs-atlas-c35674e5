# Form Components

Form components built with React Hook Form and Radix UI primitives.

## Form

Wrapper component that provides form context using React Hook Form's `FormProvider`.

### Props

Accepts all props from React Hook Form's `FormProvider`.

<ParamField path="...methods" type="UseFormReturn" required>
  Form methods from `useForm()` hook. Spread all methods as props.
</ParamField>

### Usage

```tsx
import { useForm } from "react-hook-form";
import { Form } from "@/components/ui/form";

export default function Example() {
  const form = useForm();

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)}>
        {/* Form fields */}
      </form>
    </Form>
  );
}
```

## FormField

Controller component that connects form inputs to React Hook Form.

### Props

<ParamField path="name" type="string" required>
  The name of the field in the form data.
</ParamField>

<ParamField path="control" type="Control" required>
  Form control from `useForm()` or `useFormContext()`.
</ParamField>

<ParamField path="render" type="function" required>
  Render function that receives field props. Returns the form input element.
</ParamField>

### Usage

```tsx
import { FormField, FormItem, FormLabel, FormControl } from "@/components/ui/form";
import { Input } from "@/components/ui/input";
import { useFormContext } from "react-hook-form";

export default function Example() {
  const { control } = useFormContext();

  return (
    <FormField
      control={control}
      name="email"
      render={({ field }) => (
        <FormItem>
          <FormLabel>Email</FormLabel>
          <FormControl>
            <Input type="email" {...field} />
          </FormControl>
        </FormItem>
      )}
    />
  );
}
```

## FormLabel

Label component that automatically connects to the form field and shows error state.

### Props

Accepts all Radix UI Label props.

<ParamField path="htmlFor" type="string">
  Automatically set by form context. Associates label with input.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes. Error styling is automatically applied.
</ParamField>

---

## FormControl

Wrapper that connects child input to form field with proper ARIA attributes.

### Props

Uses Radix UI's Slot component to pass props to child element.

## Complete Form Example

<Tabs>
  <Tab title="Basic Form">
    ```tsx
    import { useForm } from "react-hook-form";
    import { zodResolver } from "@hookform/resolvers/zod";
    import * as z from "zod";
    import {
      Form,
      FormField,
      FormItem,
      FormLabel,
      FormControl,
      FormDescription,
      FormMessage,
    } from "@/components/ui/form";
    import { Input } from "@/components/ui/input";
    import { Button } from "@/components/ui/button";

    const formSchema = z.object({
      email: z.string().email("Invalid email address"),
      password: z.string().min(8, "Password must be at least 8 characters"),
    });

    export default function LoginForm() {
      const form = useForm<z.infer<typeof formSchema>>({
        resolver: zodResolver(formSchema),
        defaultValues: {
          email: "",
          password: "",
        },
      });

      function onSubmit(values: z.infer<typeof formSchema>) {
        console.log(values);
      }

      return (
        <Form {...form}>
          <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
            <FormField
              control={form.control}
              name="email"
              render={({ field }) => (
                <FormItem>
                  <FormLabel>Email</FormLabel>
                  <FormControl>
                    <Input type="email" placeholder="you@example.com" {...field} />
                  </FormControl>
                  <FormDescription>
                    We'll never share your email.
                  </FormDescription>
                  <FormMessage />
                </FormItem>
              )}
            />
            <FormField
              control={form.control}
              name="password"
              render={({ field }) => (
                <FormItem>
                  <FormLabel>Password</FormLabel>
                  <FormControl>
                    <Input type="password" {...field} />
                  </FormControl>
                  <FormMessage />
                </FormItem>
              )}
            />
            <Button type="submit">Sign In</Button>
          </form>
        </Form>
      );
    }
    ```
  </Tab>
</Tabs>

## Field Components

Custom field components that provide flexible layouts and styling options.

### FieldSet

Grouping container for related form fields.

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

```tsx
import { FieldSet, FieldLegend } from "@/components/ui/field";

<FieldSet>
  <FieldLegend>Personal Information</FieldLegend>
  {/* Form fields */}
</FieldSet>
```

### FieldLegend

Legend for a fieldset.

<ParamField path="variant" type="string" default="legend">
  Style variant. Options: `legend`, `label`
</ParamField>

### FieldGroup

Container for organizing multiple fields with responsive layouts.

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

### Field

Flexible field container with orientation support.

<ParamField path="orientation" type="string" default="vertical">
  Layout orientation. Options:
  - `vertical` - Stacked layout
  - `horizontal` - Side-by-side layout
  - `responsive` - Vertical on mobile, horizontal on larger screens
</ParamField>

### FieldLabel

Label component for Field.

<ParamField path="htmlFor" type="string">
  Associates label with input element.
</ParamField>

### FieldContent

Container for field input and related elements.

### FieldTitle

Title text for a field.

### FieldDescription

Helper text for providing additional field context.

### FieldError

Error message component for field validation errors.

<ParamField path="errors" type="Array<{message?: string}>">
  Array of error objects. Automatically renders error messages.
</ParamField>

### Field Components Example

```tsx
import {
  Field,
  FieldLabel,
  FieldContent,
  FieldDescription,
  FieldError,
} from "@/components/ui/field";
import { Input } from "@/components/ui/input";

export default function Example() {
  return (
    <Field orientation="vertical">
      <FieldLabel htmlFor="username">Username</FieldLabel>
      <FieldContent>
        <Input id="username" placeholder="Enter username" />
        <FieldDescription>
          Choose a unique username for your account.
        </FieldDescription>
        <FieldError errors={[{ message: "Username is required" }]} />
      </FieldContent>
    </Field>
  );
}
```

## Select

Dropdown select component with search and grouping support.

### Components

- `Select` - Root component
- `SelectTrigger` - Button that opens the dropdown
- `SelectValue` - Displays selected value
- `SelectContent` - Dropdown content container
- `SelectItem` - Individual option
- `SelectGroup` - Group of related items
- `SelectLabel` - Label for a group
- `SelectSeparator` - Visual separator

### Props

<ParamField path="value" type="string">
  Controlled selected value.
</ParamField>

<ParamField path="onValueChange" type="(value: string) => void">
  Callback when selection changes.
</ParamField>

<ParamField path="defaultValue" type="string">
  Default selected value.
</ParamField>

<ParamField path="disabled" type="boolean">
  Disables the select.
</ParamField>

### Usage
